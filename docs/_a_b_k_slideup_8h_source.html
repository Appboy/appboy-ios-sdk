3 /*
4  * There are two possible values which control where the slideup will enter the view.
5  *
6  * ABKSlideupFromBottom - This is the default behavior.
7  * The slideup will slide onto the screen from the bottom edge of the view and will hide by sliding back down off
8  * the bottom of the screen.
9  *
10  * ABKSlideupFromTop - The slideup will slide onto the screen from the top edge of the view and will hide by sliding
11  * back up off the top of the screen.
12  */

14  ABKSlideupFromTop,

18 /*
19  * The ABKSlideupDismissType defines how the slideup can be dismissed.
20  *
21  * ABKSlideupDismissAutomatically - This is the default behavior. It will dismiss after the length of time defined by

31 /*
32  * The ABKSlideupClickActionType defines the action that will be performed when the Slideup is clicked.
33  *
34  * ABKSlideupDisplayNewsFeed - This is the default behavior. It will open a modal view of Appboy news feed.
35  *
36  * ABKSlideupRedirectToURI - The slideup will try to redirect to the uri defined by the uri property. Only when the uri
37  * is an HTTP URL, a modal web view will be displayed. If the uri is a protocol uri, the slideup will redirect to the
38  * protocol uri.
39  *
40  * ABKSlideupNoneClickAction - The slideup will do nothing but dismiss itself.
41  */
42 typedef NS_ENUM(NSInteger, ABKSlideupClickActionType) {
43  ABKSlideupDisplayNewsFeed,
44  ABKSlideupRedirectToURI,
45  ABKSlideupNoneClickAction
46 };
